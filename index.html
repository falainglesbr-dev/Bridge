// Configuração da sua nova API const API_URL = "https://script.google.com/macros/s/AKfycbwoMQolodkRhFOUhC2HQRBqRedZioxXT33dOWLSLoNcYJEJ40C2zszh54Lp8xsxcxtmXA/exec"; async function realizarLogin() { const token = document.getElementById('token-input').value; const senha = document.getElementById('senha-input').value; if (!token || !senha) { alert("Por favor, preencha todos os campos."); return; } try { const response = await fetch(`${API_URL}?action=login&token=${token}&senha=${senha}`); const data = await response.json(); if (data.success) { // SALVA AS CREDENCIAIS NO NAVEGADOR localStorage.setItem('bridge_token', data.token); localStorage.setItem('bridge_senha', senha); // Importante para as próximas requisições localStorage.setItem('bridge_user', JSON.stringify(data)); // REDIRECIONA PARA A DASHBOARD window.location.href = 'dashboard.html'; } else { alert("Erro: " + data.message); } } catch (error) { console.error("Erro no login:", error); alert("Erro ao conectar com o servidor."); } }
